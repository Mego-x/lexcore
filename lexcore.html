const histories = {};


async function sendMsg(agentId) {

  const input = document.getElementById(`input-${agentId}`);

  const text = input.value.trim();

  if (!text) return;

  input.value = '';

  await runDemo(agentId, text);

}


async function sendPrompt(agentId, prompt) {

  const row = document.getElementById(`row-${agentId}`);

  if (!row.classList.contains('expanded')) row.classList.add('expanded');

  await runDemo(agentId, prompt);

}


async function runDemo(agentId, userText) {

  const agent = agents.find(a => a.id === agentId);

  const msgsEl = document.getElementById(`msgs-${agentId}`);

  const sendBtn = document.getElementById(`send-${agentId}`);


  if (!histories[agentId]) histories[agentId] = [];

  histories[agentId].push({ role: 'user', content: userText });


  // User bubble

  msgsEl.innerHTML += `<div class="msg msg-user">${escHtml(userText)}</div>`;


  // Typing indicator

  const typingId = `typing-${agentId}-${Date.now()}`;

  msgsEl.innerHTML += `<div class="msg msg-agent msg-typing" id="${typingId}">

    <div class="typing-dot"></div><div class="typing-dot"></div><div class="typing-dot"></div>

  </div>`;

  msgsEl.scrollTop = msgsEl.scrollHeight;

  sendBtn.disabled = true;


  try {

    const res = await fetch('https://api.anthropic.com/v1/messages', {

      method: 'POST',

      headers: {

        'Content-Type': 'application/json',

        'x-api-key': API_KEY,

        'anthropic-version': '2023-06-01',

        'anthropic-dangerous-direct-browser-access': 'true'

      },

      body: JSON.stringify({

        model: 'claude-sonnet-4-5',

        max_tokens: 1000,

        system: agent.systemPrompt,

        messages: histories[agentId]

      })

    });


    const data = await res.json();

    const reply = data?.content?.[0]?.text || "I'm having trouble responding right now. Please try again.";

    histories[agentId].push({ role: 'assistant', content: reply });


    document.getElementById(typingId)?.remove();

    msgsEl.innerHTML += `<div class="msg msg-agent">${formatReply(reply)}</div>`;

  } catch(e) {

    document.getElementById(typingId)?.remove();

    msgsEl.innerHTML += `<div class="msg msg-agent">⚠️ Connection error. Please check your network and try again.</div>`;

  }


  sendBtn.disabled = false;

  msgsEl.scrollTop = msgsEl.scrollHeight;

}

}

function formatReply(t) {

  return escHtml(t)

    .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')

    .replace(/\n/g, '<br/>');

}

function formatCard(el) {

  let v = el.value.replace(/\D/g,'').substring(0,16);

  el.value = v.replace(/(.{4})/g,'$1 ').trim();

}

function formatExp(el) {

  let v = el.value.replace(/\D/g,'').substring(0,4);

  if (v.length >= 2) v = v.substring(0,2) + ' / ' + v.substring(2);

  el.value = v;

}
